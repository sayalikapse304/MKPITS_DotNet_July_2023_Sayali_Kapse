using System;
namespace program
{
    class swapping
    {
        public static void Main()
        {
            int num1 = 5, num2 = 6, add;
            add = num1;
            num1 = num2;
            num2 = add;
            Console.WriteLine("After Swapping");
            Console.WriteLine("num1 =" + num1);
            Console.WriteLine("num2 =" + num2);
            Console.ReadLine();
        }
    }

}
